--- validation.js document.getElementById("registrationForm").addEventListener("submit", function (event) { event.preventDefault(); const firstName = document.getElementById("firstName").value.trim(); if (!/^[a-zA-Z]{6,}$/.test(firstName)) { alert("Minimum 6 characters required for First Name and only alphabets allowed"); return; } const email = document.getElementById("email").value.trim(); if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)) { alert("Enter a valid email"); return; } const mobile = document.getElementById("mobile").value.trim(); if (!/^\d{10}$/.test(mobile)) { alert("Mobile number must be exactly 10 digits"); return; } alert("Registration successful!"); document.getElementById("registrationForm").submit(); });
